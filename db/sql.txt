DROP TABLE IF EXISTS account;
DROP TABLE IF EXISTS entry;

CREATE TABLE account(
	id SERIAL PRIMARY KEY,
	login TEXT UNIQUE,
	encrypted_control_string BYTEA,
	salt BYTEA
);

CREATE TABLE entry(
	id SERIAL PRIMARY KEY,
	user_login TEXT REFERENCES account(login),
	service_name TEXT,
	login TEXT,
	encrypted_password BYTEA,
	salt BYTEA
);
