CREATE TABLE IF NOT EXISTS account(
	id SERIAL PRIMARY KEY,
	login TEXT UNIQUE,
	encrypted_control_string BYTEA,
	salt BYTEA
);

CREATE TABLE IF NOT EXISTS entry(
	id SERIAL PRIMARY KEY,
	user_login TEXT REFERENCES account(login),
	service_name TEXT,
	login TEXT,
	encrypted_password BYTEA,
	salt BYTEA
);
